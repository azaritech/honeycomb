    public cubeToCartesian(cubeCoordinates: Object): Object source 

    import {cubeToCartesian} from 'honeycomb-grid/src/hex/prototype.js'

    Params:

    NameTypeAttributeDescription
    cubeCoordinates Object

    At least the q and r cube coordinates.
    cubeCoordinates.q number

    The q cube coordinate.
    cubeCoordinates.r number

    The r cube coordinate.
    cubeCoordinates.s number
    • optional

    The optional s cube coordinate.

    Return:

    Object

    The hex's cartesian x and y coordinates.

    Example:

    const Hex = Honeycomb.extendHex()

Hex().cubeToCartesian({ q: 1, r: 2, s: -3 }) // { x: 2, y: 2 }
// the `s` coordinate isn't required:
Hex().cubeToCartesian({ q: -3, r: 5 })       // { x: -1, y: 5 }

    public get(keyOrPoint: number | point): hex source 

    import {get} from 'honeycomb-grid/src/grid/prototype.js'

    Get a hex from a grid.

    Params:

    NameTypeAttributeDescription
    keyOrPoint number | point

    An index/key or a point.

    Return:

    hex

    The found hex or undefined.

    Example:

    const Grid = Honeycomb.defineGrid()
const Hex = Grid.Hex
const grid = Grid.rectangle({ width: 2, height: 2 })

grid.get(0)              // { x: 0, y: 0 }
grid.get(Hex(0, 1))      // { x: 0, y: 1 }
grid.get({ x: 0, y: 1 }) // { x: 0, y: 1 }
grid.get([0, 1])         // { x: 0, y: 1 }

grid.get(42)             // undefined
grid.get(Hex(6, -2))     // undefined

    public hexesBetween(firstHex: hex, lastHex: hex): hex[] source 

    import {hexesBetween} from 'honeycomb-grid/src/grid/prototype.js'

    Params:

    NameTypeAttributeDescription
    firstHex hex

    The first hex.
    lastHex hex

    The last hex.

    Return:

    hex[]

    Array (not a grid) of hexes in a straight line from firstHex to (and including) lastHex.

    Example:

    const Grid = Honeycomb.defineGrid()
const Hex = Grid.Hex
const grid = Grid.rectangle({ width: 4, height: 4 })

grid.hexesBetween(Hex(), Hex(3)) // [
                                 //    { x: 0, y: 0 },
                                 //    { x: 0, y: 1 },
                                 //    { x: 1, y: 1 },
                                 //    { x: 2, y: 2 },
                                 //    { x: 3, y: 2 },
                                 //    { x: 3, y: 3 },
                                 // ]

    See:

    • {@link https://www.redblobgames.com/grids/hexagons/#line-drawing|redblobgames.com}
